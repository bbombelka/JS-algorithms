    <div id="filter" class="visible">filter blablab</div>

    <script>
      const selection = document.querySelector("select");
      const messages = [...document.querySelectorAll("div")];
      let selected;

      function displayConsoleOne() {
        console.log("funkcja 1");
      }

      function displayConsoleTwo() {
        console.log("funkcja 2");
      }

      function displayConsoleThree() {
        console.log("funkcja 3");
      }
      const functions = {
        filter: displayConsoleOne,
        slice: displayConsoleTwo,
        regex: displayConsoleThree
      };

      selection.addEventListener("change", e => {
        messages
          .filter(div => div.classList.contains("visible"))[0]
          .classList.remove("visible");

        messages
          .filter(div => div.id === e.target.value)[0]
          .classList.add("visible");

        selected = e.target.value;
      });

      function init() {
        return functions[selected]();
      }

      document.addEventListener("click", init);

      //   let selected = selection.filter(option => option.selected === true)[0].value;
    </script>
